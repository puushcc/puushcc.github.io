function deepCopy(obj, targetObj) {
  for (let key in obj) {
    let item = obj[key];
    if (item instanceof Array) {//if array
      targetObj[key] = [];
      deepCopy(item, targetObj[key]);
    } else if (item instanceof Object) {//if object
      targetObj[key] = {};
      deepCopy(item, targetObj[key]);
    } else {//normal attribute
      targetObj[key] = obj[key];
    }
  }
}

/**
* 获取数据源所有数据
* @theDataSourse 数据源
* @someJson  字符串类型的值 如：const someJson = {"textField_kkm9o5cd":"123"}
*
*/
export function getAllData(theDataSourse, someJson){
  //返回的所有数据
  var theDataSourseAllData = {}
  //// 如果直接把 someJson 传给 searchFieldJson 是不会有效果的。我们需要通过 JSON.stringify 方法，把这个对象转成字符串形式。
  const fieldJson = JSON.stringify(someJson);
  //获取数据，最大值为100
  this.dataSourceMap[theDataSourse].load({
    _csrf_token: csrfToken, 
    searchFieldJson: fieldJson,   //查询条件
    currentPage: 1,
    pageSize: 100
  }).then(res => {
    //res： 蒂一页100条内的数据
    //深拷贝
    deepCopy(res, theDataSourseAllData);
    //获取查询的总条数
    var totalCount = res.totalCount
    //判断查询所有数据的总页码数
    var lening = parseInt(totalCount / 100)
    //如果查询的总条数 > 100 继续查询
    if (lening > 0) {
      //遍历查询所有数据
      for (let j = 2; j < lening + 2; j++) {
        this.dataSourceMap[theDataSourse].load({
          _csrf_token: csrfToken,
          searchFieldJson: fieldJson,   //查询条件
          currentPage: j,
          pageSize: 100
        }).then(res => {
          //res： 蒂j页100条内的数据
          // console.log('theDataSourseAllData.data', theDataSourseAllData.data)
          theDataSourseAllData.data = theDataSourseAllData.data.concat(res.data);
        })
      }
    }
    Promise.all(theDataSourseAllData.data).then(res => {
      // console.log('requestArrRes', res)
      console.log('theDataSourseAllData', theDataSourseAllData)
      this.setState({
        theDataSourseAllData: theDataSourseAllData
      });
      return theDataSourseAllData;
    })
  })
}

function formatDate(date) {
  var date = new Date(date);
  var YY = date.getFullYear() + '-';
  var MM = (date.getMonth() + 1 < 10 ? '0' + (date.getMonth() + 1) : date.getMonth() + 1) + '-';
  var DD = (date.getDate() < 10 ? '0' + (date.getDate()) : date.getDate());
  var hh = (date.getHours() < 10 ? '0' + date.getHours() : date.getHours()) + ':';
  var mm = (date.getMinutes() < 10 ? '0' + date.getMinutes() : date.getMinutes()) + ':';
  var ss = (date.getSeconds() < 10 ? '0' + date.getSeconds() : date.getSeconds());
  return YY + MM + DD;
}
